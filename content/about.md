<script>
document.getElementById("generate-qr").addEventListener("click", function () {
  const amountInput = document.getElementById("donation-amount");
  const amount = amountInput.value;
  const generateBtn = this;
  
  if (!amount || isNaN(amount) || amount <= 0) {
    amountInput.classList.add("is-invalid");
    setTimeout(() => { amountInput.classList.remove("is-invalid"); }, 1500);
    return;
  }

  const qrContainer = document.getElementById("qrcode-container");
  const loader = document.getElementById("qr-loading");
  
  // Reset State
  qrContainer.innerHTML = "";
  qrContainer.classList.add("d-none");
  qrContainer.classList.add("flex-column"); // Ensure items stack vertically
  
  // Show Loader
  loader.classList.remove("d-none");
  loader.classList.add("d-flex");
  generateBtn.disabled = true;
  generateBtn.innerHTML = `<span class="spinner-border spinner-border-sm" role="status" aria-hidden="true"></span>`;

  // Simulated Delay
  setTimeout(() => {
    loader.classList.remove("d-flex");
    loader.classList.add("d-none");
    qrContainer.classList.remove("d-none");
    
    generateBtn.disabled = false;
    generateBtn.innerHTML = `<i class="fa-solid fa-qrcode"></i> Generate`;

    const upiId = "9362657944@slc";
    const name = "Lucky Taorem";
    const upiUrl = `upi://pay?pa=${upiId}&pn=${encodeURIComponent(name)}&am=${amount}&cu=INR`;

    // 1. Create outer structural wrapper
    const wrapper = document.createElement("div");
    wrapper.className = "qr-wrapper";
    
    // 2. Create inner target specifically for QRCode.js
    const qrTarget = document.createElement("div");
    wrapper.appendChild(qrTarget);

    // 3. Generate colored QR code
    new QRCode(qrTarget, {
      text: upiUrl,
      width: 200,
      height: 200,
      colorDark: "#1f1d2b",
      colorLight: "#ffffff",
      correctLevel: QRCode.CorrectLevel.H
    });

    // 4. Create Logo Overlay
    const logo = document.createElement("div");
    logo.className = "qr-logo-overlay";
    logo.innerHTML = `<img src="https://ltdeveloperblogs.github.io/img/logo512x512.png" alt="Logo" style="width: 100%; height: 100%; object-fit: contain; border-radius: 8px;">`;
    wrapper.appendChild(logo);

    // 5. Append QR wrapper to screen
    qrContainer.appendChild(wrapper);

    // 6. Create and append the beautifully animated Thank You message
    const thankYouMsg = document.createElement("div");
    thankYouMsg.className = "mt-4 text-center";
    // Reuse the popIn animation from CSS, slightly delayed
    thankYouMsg.style.animation = "popIn 0.5s 0.35s cubic-bezier(0.175, 0.885, 0.32, 1.275) both";
    thankYouMsg.innerHTML = `
      <h5 class="text-success fw-bold mb-1">❤️ Thank You!</h5>
      <p class="text-body-secondary small mb-0">Scan to complete your ₹${amount} contribution.</p>
    `;
    qrContainer.appendChild(thankYouMsg);

    // 7. Show Toast
    const toastEl = document.getElementById("thankyou-toast");
    const toast = new bootstrap.Toast(toastEl);
    toast.show();
    
  }, 800);
});
</script>
